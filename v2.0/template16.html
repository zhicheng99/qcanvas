拖动限制范围 (现只支持Qarc、Qpolygon、Qrect实例)
dragRange格式 [[左上角x,左上角y]，[右下角x,右下角y]]
var qcanvas = new Qcanvas({
    id: 'qcanvas',
    width: 500,
    height: 500,
});

var rect = qcanvas.rect({
    start: [160, 100],
    width: 100,
    height: 100,
    borderColor: 'red',
    fillColor: '',
    // degree:50,
    drag: false
})

var arc = qcanvas.arc({
    start: [200, 120],
    sAngle: 0,
    eAngle: 360,
    borderColor: 'blue',
    fillColor: '',
    r: 10,
    // drag:'vertical',
    dragRange: [rect.start, [rect.start[0] + rect.width, rect.start[1] + rect.height]]

})

var polygon = qcanvas.polygon({
    r: 15,
    num: 6,
    start: [180, 120],
    opacity: 0.4,
    dragRange: [rect.start, [rect.start[0] + rect.width, rect.start[1] + rect.height]]
})

var rect = qcanvas.rect({
    start: [170, 120],
    width: 20,
    height: 20,
    borderColor: 'green',
    fillColor: '',
    dragRange: [rect.start, [rect.start[0] + rect.width, rect.start[1] + rect.height]]

})
