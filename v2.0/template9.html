var qcanvas = new Qcanvas({
	 id: 'qcanvas',
        width: 300,
        height: 200
});
var rect = qcanvas.rect({
    start: [0, 10],
    borderColor: 'red',
    fillColor: 'blue',
    drag: false
})

qcanvas.qanimation.animate(rect, {
    start: [0, 10],
}, {
    start: [200, 10],
}, 2, true, 'Back.easeInOut');
