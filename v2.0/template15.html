var qcanvas = new Qcanvas(
		{
			id:'qcanvas',
			width:500,
			height:500,
			mousedown:function(){
				console.log('canvas mousedown')
				console.log(qcanvas);
				if(qcanvas.qresize !== null){

					qcanvas.qresize.resizeLayer.setDisplay('none');

				}

				if(qcanvas.qrotate !== null){

					qcanvas.qrotate.rotateLayer.setDisplay('none');

				}
			}
		});

var rect = qcanvas.qrect.rect({
