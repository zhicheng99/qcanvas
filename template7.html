var qcanvas = new Qcanvas(["qcanvas",300,200]);

qcanvas.load({
	"b":"img/img.png"
},function(){ 
	var b = qcanvas.getSourceByName("b");
	
	
	qcanvas.qimg.img({
		img:'img/img.png',
		sStart:[0,0],
		sWidth:b.width,
		sHeight:b.height,
		size:"cover",
		tStart:[100,0],
		tWidth:50,
		tHeight:120,
		degree:30,
		mousemove:function(position){
			console.log(position);
		}
		
		
	});
});
